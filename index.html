    weui.datePicker({
        id: "start" + id,
        start: 2000,
        end: dt.getFullYear() + 20,
        defaultValue: date,
        onConfirm: function(result) {
            $("#datetime-picker").val(result[0].label.replace("年", "-") + result[1].label.replace("月", "-") + result[2].label

        }
    });
